Gráfico com ggplot2

dados_starwars %>% 
  ggplot() +
  geom_point(aes(x = massa, y = altura)) +
  theme_light() + 
  gghighlight::gghighlight(nome == "Darth Vader", label_key = nome) +
  labs(x = "Massa", y = "Altura")
